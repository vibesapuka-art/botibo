async function iniciar() { const mac = document.getElementById('mac').value; // ... restante do código de envio via fetch ... const monitor = setInterval(async () => { const res = await fetch(`/status?mac=${mac}`); const data = await res.json(); // LOGICA DE ERRO NO PAINEL if (data.status === "erro_login") { clearInterval(monitor); // Estilização de Erro const statusTxt = document.getElementById('main-status'); const barCheck = document.getElementById('bar-check'); statusTxt.innerText = "❌ DADOS INCORRETOS!"; statusTxt.style.color = "#ff4444"; // Vermelho barCheck.style.width = "100%"; barCheck.style.background = "#ff4444"; alert("Ocorreu um erro: MAC Address ou Device Key são inválidos. Verifique os dados no seu aplicativo e tente novamente."); return; } // LOGICA DE SUCESSO if (data.status === "processando" || data.status === "ok") { document.getElementById('status-check').style.display = 'none'; document.getElementById('status-play').style.display = 'block'; const p = (data.concluidos / data.total) * 100; document.getElementById('bar-play').style.width = p + "%"; document.getElementById('main-status').innerText = `Processando: ${data.concluidos}/${data.total}`; if (data.status === "ok") { clearInterval(monitor); document.getElementById('main-status').innerText = "✅ FINALIZADO COM SUCESSO!"; } } }, 4000); }
